Demo angular-bsModal

Modal with description and confirm callback

This is simple modal with description and confirm function callback. 

              
                bsModal.show({
                  title       : 'First Modal Example',
                  description : 'This is descripton in first modal example. Do You agreed?',
                  confirmText : 'Yes!',
                  confirm     : function() {
                    bsModal.close();
                  }     
                });

Change width and background color

Add "width" and "color" to injected object in bsModal to change width and/or background.
To maintain responsiveness set width always in percent. 

              
                bsModal.show({
                  title       : 'Change the width and background',
                  description : 'Modal with 90% width and pink backgorund',
                  width       : '90%',
                  background  : '#ff80ab',
                  confirmText : 'Cool',
                  confirm     : function() {
                    bsModal.close();
                  }    
                });

Modal with template

To load template into modal windows just add 'template' and path to html file

              
                bsModal.show({
                  title       : 'Modal with template',
                  template    : '../demo/modal_template/tmp.html',
                  controller  : 'modalController',
                  confirmText : 'Ok',
                  confirm     : function() {
                    bsModal.close();
                  }    
                });

Setting default options

You can set default options that will be used in each modal window. To do so, pass the default settings to bsModalProvider.

              
                angular.module('DemoProvider', ['angular-bsModal'])
                  .config(['bsModalProvider', function(bsModalProvider) {
                    // set the provider
                    bsModalProvider.setDefault({
                      title       : 'Default title',
                      description : 'This is default description',
                      width       : '20%'
                    }); 
                  }])
                  .controller('defaultController', defaultController);
                
                // and now in the controller  
                function defaultController($scope, bsModal) {
                  $scope.showModal = function() {
                    bsModal.show({
                      confirmText : 'Ok!',
                      confirm     : function() {
                        bsModal.close();
                      } 
                    });      
                  };
                };

Set alias

You can set multiple aliases to call them by name. To do this, create an alias using the method of " createAlias ​​()" with provider
Second argument in showAs() method is optional. When You want change some defined option simply add second argument with options object

              
                angular.module('DemoProvider', ['angular-bsModal'])
                  .config(['bsModalProvider', function(bsModalProvider) {
                    // set the provider
                    bsModalProvider.setDefault({
                      title       : 'Default title',
                      description : 'This is default description',
                      width       : '20%'
                    });
                    
                    bsModalProvider.createAlias('sessionExpired', {
                      title       : 'Session expired!',
                      description : 'Your session has expired. Please login again',
                      width       : '20%',
                      close       : false,
                      confirmText : 'Go to login page',
                      confirm     : function() {
                        bsModalProvider.close();
                      }
                    }) 
                  }])
                  .controller('aliasController', aliasController)
                
                // and now in the controller              
                function aliasController($scope, bsModal) {
                  $scope.showAliasModal = function() {
                    bsModal.showAs('sessionExpired');  
                  };
                  
                  $scope.showAliasModalWithDescription = function() {
                    bsModal.showAs('sessionExpired', {
                      description : 'Description from controller'
                    });  
                  };  
                };
